@router.post("/decoys/generate")
async def generate_decoys_ai(
    request: DecoyGenerationRequest,
    session: AsyncSession = Depends(get_session),
) -> AttackResponseContainer:
    """
    Generate decoy assets using AI-assisted profile analysis.
    
    Request: DecoyGenerationRequest (site URL, description, etc.)
    Response: Array of DecoyAssetCreateResponse (same schema as manual creation)
    
    Uses app.ai.generate_site_profile() and app.ai.generate_decoy_assets() internally.
    If LLM not available, falls back to default decoys.
    """
    logger.info(f"[DECOY_API] Generating decoys for {request.site_url}")
    
    try:
        # Import here to avoid circular deps
        from app.ai import generate_site_profile, generate_decoy_assets
        
        # 1. Profile the site
        site_profile = generate_site_profile(
            url=request.site_url,
            page_title=request.page_title,
            description=request.description,
        )
        logger.info(f"[DECOY_API] Generated profile: {site_profile.site_name}")
        
        # 2. Generate decoy assets
        decoy_defs = generate_decoy_assets(site_profile, count=request.count)
        logger.info(f"[DECOY_API] Generated {len(decoy_defs)} decoys")
        
        # 3. Save each decoy to DB
        results = []
        for decoy_def in decoy_defs:
            decoy_asset = DecoyAsset(
                id=str(uuid4()),
                asset_type=decoy_def.asset_type,
                name=decoy_def.name,
                path=decoy_def.path,
                description=decoy_def.description,
            )
            session.add(decoy_asset)
            
            result = DecoyAssetCreateResponse(
                id=decoy_asset.id,
                name=decoy_asset.name,
                asset_type=decoy_asset.asset_type,
                path=decoy_asset.path,
            )
            results.append(result)
        
        await session.commit()
        logger.info(f"[DECOY_API] Saved {len(results)} decoys to DB")
        
        return {
            "decoys": results,
            "total": len(results),
        }
    
    except Exception as e:
        logger.error(f"[DECOY_API] Error: {e}", exc_info=True)
        await session.rollback()
        raise HTTPException(status_code=500, detail=f"Decoy generation failed: {e}")
```

**Why this works:**
- New endpoint: `/api/decoys/generate` (doesn't change existing POST /api/decoys)
- Same response schema: Array of DecoyAssetCreateResponse (frozen contract)
- Transparent AI use: If LLM available, uses it; if not, falls back to defaults
- DB integration: Saves generated decoys to the decoy_assets table

@router.get("/attacks/{session_id}")
async def get_attack_detail(
    session_id: str,
    session: AsyncSession = Depends(get_session),
    include_forensics: bool = False,  # Query param to enable forensics
) -> AttackDetailResponse:
    """
    Get full request chain for a session with optional forensic analysis.
    
    Query Parameters:
    - include_forensics (bool): If true, analyze session and return ForensicSummary
    
    Response: AttackDetailResponse (session + requests + optional forensics)
    """
    # Query by string ID (DB stores as String(36), not UUID)
    stmt = select(Session).where(Session.id == session_id)
    result = await session.execute(stmt)
    sess = result.scalar_one_or_none()
    
    if not sess:
        raise HTTPException(status_code=404, detail=f"Session {session_id} not found")
    
    sess_summary = SessionSummarySchema.from_orm(sess)
    requests = [RequestInfoSchema.from_orm(r) for r in sess.requests]
    
    forensics = None
    if include_forensics:
        try:
            from app.ai import summarize_session
            
            logger.info(f"[ATTACK_API] Analyzing session {session_id} for forensics")
            
            # Convert request ORM objects to dicts for AI
            request_dicts = [
                {
                    "method": r.method,
                    "path": r.path,
                    "query_string": r.query_string,
                    "body": r.body[:500] if r.body else None,  # Truncate large bodies
                }
                for r in sess.requests
            ]
            
            forensic_summary = summarize_session(
                session_id=session_id,
                requests=request_dicts,
                ip_address=sess.ip_address,
                tags=sess.tags.split(",") if sess.tags else [],
            )
            
            forensics = ForensicSummarySchema(
                headline=forensic_summary.headline,
                description=forensic_summary.description,
                techniques=forensic_summary.techniques,
                confidence=forensic_summary.confidence,
                recommendations=forensic_summary.recommendations,
            )
            logger.info(f"[ATTACK_API] Forensic analysis complete")
        except Exception as e:
            logger.error(f"[ATTACK_API] Forensics analysis failed: {e}")
            # Don't fail the whole request, just omit forensics
            forensics = None
    
    return AttackDetailResponse(
        session=sess_summary,
        requests=requests,
        forensics=forensics,
    )
```

**Why this works:**
- Backward compatible: `forensics` field is optional; existing clients still work
- Query parameter: `?include_forensics=true` to opt-in
- Graceful degradation: If AI fails, forensics is null; request succeeds
- Performance: Only runs forensic analysis if explicitly requested

class TestAIIntegration:
    """Test AI-assisted endpoints."""
    
    async def test_generate_decoys_ai(self, client):
        """POST /api/decoys/generate should generate and save decoys."""
        response = await client.post(
            "/api/decoys/generate",
            json={
                "site_url": "https://example.com",
                "page_title": "Example Corp",
                "tech_stack": ["nginx", "php"],
                "description": "Business website",
                "count": 3,
            },
        )
        assert response.status_code == 200
        data = response.json()
        assert "decoys" in data
        assert len(data["decoys"]) <= 3
        
        # Verify decoys were saved to DB
        get_response = await client.get("/api/decoys")
        assert get_response.status_code == 200
    
    async def test_attack_detail_with_forensics(self, client):
        """GET /api/attacks/{id}?include_forensics=true should return forensic summary."""
        # First, create a session by hitting a honeypot
        honeypot_response = await client.get("/admin")
        assert honeypot_response.status_code == 200
        
        # Get the session list
        list_response = await client.get("/api/attacks")
        assert list_response.status_code == 200
        data = list_response.json()
        
        if data["attacks"]:
            session_id = data["attacks"][0]["id"]
            
            # Get detail WITH forensics
            detail_response = await client.get(
                f"/api/attacks/{session_id}?include_forensics=true"
            )
            assert detail_response.status_code == 200
            detail = detail_response.json()
            
            assert "session" in detail
            assert "requests" in detail
            # forensics may be null if LLM not configured
            if detail.get("forensics"):
                assert "headline" in detail["forensics"]
                assert "techniques" in detail["forensics"]
