```python
"""Site profile generation from raw signals."""

import logging
from typing import List, Dict, Optional
from app.ai.schemas import SiteProfile
from app.ai.llm_client import get_llm_client

logger = logging.getLogger(__name__)


def generate_site_profile(
    url: str,
    page_title: Optional[str] = None,
    headers: Optional[Dict[str, str]] = None,
    visible_sections: Optional[List[str]] = None,
    description: Optional[str] = None,
) -> SiteProfile:
    """
    Generate a SiteProfile from raw signals (URL, page content, headers, etc.).
    
    Uses LLM to infer site theme, tech stack, important paths, and response patterns.
    
    Args:
        url: Target URL (e.g., "https://example.com")
        page_title: HTML title tag (e.g., "WordPress Blog")
        headers: HTTP response headers (dict)
        visible_sections: Text snippets from page (e.g., ["Copyright 2024", "Powered by nginx"])
        description: Optional textual description of the site
    
    Returns:
        SiteProfile with inferred theme, tech stack, etc.
    
    Raises:
        ValueError: If LLM call fails and no fallback available
    """
    logger.info(f"[SITE_PROFILER] Profiling {url}")
    
    # Build signal summary for LLM
    signals = f"""
    URL: {url}
    Page Title: {page_title or 'N/A'}
    Headers: {headers or {}}
    Visible Sections: {visible_sections or []}
    Description: {description or 'N/A'}
    """
    
    system_prompt = """You are a web security analyst. Given raw signals about a web application, 
    profile it to identify: site name, theme (blog, ecommerce, cms, etc.), tech stack (nginx, php, mysql, etc.), 
    and important paths. Return a JSON SiteProfile."""
    
    user_prompt = f"""Profile this web application:
    {signals}
    
    Return JSON (no markdown, just raw JSON):
    {{
        "site_name": "...",
        "site_theme": "...",
        "tech_stack": [...],
        "important_paths": [...],
        "typical_response_patterns": {{}},
        "description": "..."
    }}
    """
    
    try:
        llm = get_llm_client()
        response = llm.call_json(
            system_prompt=system_prompt,
            user_prompt=user_prompt,
            response_schema=SiteProfile,
            temperature=0.3,  # Deterministic
        )
        profile = SiteProfile(**response)
        logger.info(f"[SITE_PROFILER] Generated profile: {profile.site_name}")
        return profile
    except Exception as e:
        logger.error(f"[SITE_PROFILER] LLM call failed: {e}")
        # Fallback: Basic profile from URL
        site_name = url.split("/")[2].replace("www.", "")
        return SiteProfile(
            site_name=site_name,
            site_theme="generic",
            tech_stack=[],
            important_paths=["/"],
            typical_response_patterns={
                "404": "Not Found",
                "403": "Forbidden",
            },
            description=description or "Auto-generated fallback profile",
        )
```

```python
import logging
from typing import List
from app.ai.schemas import SiteProfile, DecoyAssetDefinition
from app.ai.llm_client import get_llm_client

logger = logging.getLogger(__name__)


def generate_decoy_assets(
    site_profile: SiteProfile,
    count: int = 5,
) -> List[DecoyAssetDefinition]:
    """
    Generate realistic decoy assets for a site profile.
    
    Uses LLM to propose fake endpoints, files, and credentials based on the site's theme,
    tech stack, and typical paths.
    
    Args:
        site_profile: SiteProfile from generate_site_profile()
        count: Number of decoys to generate (default 5)
    
    Returns:
        List of DecoyAssetDefinition objects (endpoints, files, credentials)
    
    Raises:
        ValueError: If LLM call fails and no fallback available
    """
    logger.info(f"[DECOY_GEN] Generating {count} decoys for {site_profile.site_name}")
    
    system_prompt = """You are a honeypot design expert. Given a site profile, generate realistic decoy assets 
    that would attract attackers but not interfere with legitimate traffic. Return JSON array only."""
    
    user_prompt = f"""Generate {count} decoy assets for this site:
    
    Site: {site_profile.site_name}
    Theme: {site_profile.site_theme}
    Tech Stack: {site_profile.tech_stack}
    Important Paths: {site_profile.important_paths}
    
    Return a JSON array (no markdown):
    [
      {{
        "asset_type": "endpoint|file|credential|api",
        "name": "unique_name",
        "path": "/admin or null for files",
        "description": "What it mimics",
        "mimics": "target_service",
        "response_template": null,
        "is_active": true
      }},
      ...
    ]
    """
    
    try:
        llm = get_llm_client()
        # TODO: Implement actual LLM call to parse array response
        # For now, return fallback decoys
        logger.warning("[DECOY_GEN] LLM not configured, using fallback decoys")
        decoys = _fallback_decoys(site_profile, count)
        return decoys
    except Exception as e:
        logger.error(f"[DECOY_GEN] LLM call failed: {e}")
        return _fallback_decoys(site_profile, count)
```

```python
"""Fake response body generation for decoy endpoints."""

import logging
from typing import Optional
from app.ai.schemas import DecoyAssetDefinition, SiteProfile
from app.ai.llm_client import get_llm_client

logger = logging.getLogger(__name__)


def generate_response_body(
    decoy: DecoyAssetDefinition,
    site_profile: SiteProfile,
    method: str = "GET",
    request_headers: Optional[dict] = None,
) -> str:
    """
    Generate a realistic fake response body for a decoy endpoint.
    
    Args:
        decoy: DecoyAssetDefinition (what type of response to generate)
        site_profile: SiteProfile (for context and realism)
        method: HTTP method (GET, POST, etc.)
        request_headers: Optional incoming request headers (for Accept, etc.)
    
    Returns:
        Response body as string (HTML, JSON, plain text, etc.)
    
    Raises:
        ValueError: If response generation fails
    """
    logger.info(f"[RESPONSE_WRITER] Generating response for {decoy.name}")
    
    system_prompt = """You are a web server simulator. Given a decoy endpoint type and site context,
    generate a realistic HTTP response body (HTML, JSON, plain text, etc.) that would fool an attacker
    into thinking they found a real vulnerability or leaked file."""
    
    user_prompt = f"""Generate a {method} response for this fake endpoint:
    
    Site: {site_profile.site_name}
    Decoy Type: {decoy.asset_type}
    Mimics: {decoy.mimics}
    Path: {decoy.path or '(file)'}
    Description: {decoy.description}
    
    Return ONLY the response body (no headers, no markdown). If JSON, return valid JSON. If HTML, return valid HTML.
    """
    
    try:
        llm = get_llm_client()
        # TODO: Implement actual LLM call
        logger.warning("[RESPONSE_WRITER] LLM not configured, using fallback response")
        return _fallback_response(decoy, site_profile)
    except Exception as e:
        logger.error(f"[RESPONSE_WRITER] LLM call failed: {e}")
        return _fallback_response(decoy, site_profile)


def _fallback_response(decoy: DecoyAssetDefinition, site_profile: SiteProfile) -> str:
    """Generate basic fallback response if LLM unavailable."""
    if decoy.asset_type == "endpoint":
        return f"""<!DOCTYPE html>
<html>
<head><title>{site_profile.site_name} - {decoy.mimics}</title></head>
<body>
<h1>{decoy.mimics}</h1>
<p>Welcome to {site_profile.site_name}</p>
</body>
</html>
"""
    elif decoy.asset_type == "file":
        if decoy.mimics == "env_file":
            return f"""DATABASE_URL=postgresql://user:password@localhost/dbname
API_KEY=sk-1234567890abcdef
SECRET=super-secret-key-here
"""
        return f"# Fake {decoy.mimics} file\n"
    else:
        return '{"status": "ok", "data": []}'
```

```python
"""Forensic session analysis and summary generation."""

import logging
from typing import List, Dict, Optional
from app.ai.schemas import ForensicSummary
from app.ai.llm_client import get_llm_client

logger = logging.getLogger(__name__)


def summarize_session(
    session_id: str,
    requests: List[Dict],
    tags: Optional[List[str]] = None,
    ip_address: Optional[str] = None,
) -> ForensicSummary:
    """
    Analyze a honeypot session and generate a forensic summary.
    
    Args:
        session_id: Session ID (for reference)
        requests: List of request dicts (method, path, query_string, body, etc.)
        tags: Optional pre-assigned tags (e.g., ["scanner", "crawler"])
        ip_address: Attacker IP address
    
    Returns:
        ForensicSummary with headline, narrative, techniques, confidence, recommendations
    
    Raises:
        ValueError: If analysis fails
    """
    logger.info(f"[FORENSIC] Analyzing session {session_id}")
    
    # Summarize request patterns
    request_summary = _summarize_requests(requests)
    
    system_prompt = """You are a cybersecurity forensics analyst. Given request logs from a honeypot,
    analyze the attacker behavior and provide a brief forensic summary: headline, narrative, 
    suspected attack techniques, confidence level (0.0-1.0), and recommendations."""
    
    user_prompt = f"""Analyze this honeypot session:
    
    Session ID: {session_id}
    IP: {ip_address or 'unknown'}
    Tags: {tags or []}
    
    Request Summary:
    {request_summary}
    
    Return JSON (no markdown):
    {{
        "headline": "One-liner assessment",
        "description": "2-3 sentence narrative",
        "techniques": ["technique1", "technique2"],
        "confidence": 0.85,
        "recommendations": ["action1", "action2"]
    }}
    """
    
    try:
        llm = get_llm_client()
        # TODO: Implement actual LLM call
        logger.warning("[FORENSIC] LLM not configured, using fallback summary")
        return _fallback_summary(session_id, request_summary)
    except Exception as e:
        logger.error(f"[FORENSIC] LLM call failed: {e}")
        return _fallback_summary(session_id, request_summary)


def _summarize_requests(requests: List[Dict]) -> str:
    """Create a brief summary of requests for LLM analysis."""
    if not requests:
        return "No requests recorded."
    
    paths = [r.get("path", "?") for r in requests]
    methods = set(r.get("method", "?") for r in requests)
    
    return f"{len(requests)} requests: {', '.join(methods)} to {', '.join(set(paths[:5]))}"


def _fallback_summary(session_id: str, request_summary: str) -> ForensicSummary:
    """Generate fallback summary if LLM unavailable."""
    return ForensicSummary(
        headline="Automated scanning activity detected",
        description=f"Session {session_id} shows signs of automated reconnaissance. {request_summary}",
        techniques=["automated_scanning", "directory_enumeration"],
        confidence=0.65,
        recommendations=["Review logs", "Consider blocking IP", "Monitor for follow-up attacks"],
        raw_session_id=session_id,
    )
```
