"""LLM API client for AI integration."""

import logging
import json
from typing import Dict, Any, Optional
from pydantic import BaseModel

logger = logging.getLogger(__name__)


class LLMClient:
    """
    Wrapper for external LLM API (e.g., OpenAI, Anthropic).
    
    Provides:
    - Structured JSON output requests (auto-retry if invalid)
    - API key management from settings
    - Logging and error handling
    """
    
    def __init__(self, api_key: str, model: str = "gpt-4", base_url: Optional[str] = None):
        """
        Initialize LLM client.
        
        Args:
            api_key: API key from settings.llm_api_key
            model: Model identifier (default "gpt-4")
            base_url: Optional custom endpoint (for local models, etc.)
        """
        self.api_key = api_key
        self.model = model
        self.base_url = base_url
        logger.info(f"[LLM] Initialized with model={model}")
    
    def call_json(
        self,
        system_prompt: str,
        user_prompt: str,
        response_schema: BaseModel,
        temperature: float = 0.7,
        max_retries: int = 2,
    ) -> Dict[str, Any]:
        """
        Call LLM with structured JSON output request.
        
        Args:
            system_prompt: System context
            user_prompt: User query
            response_schema: Pydantic model for validation
            temperature: Creativity (0.0 = deterministic, 1.0 = creative)
            max_retries: If JSON invalid, retry parsing
        
        Returns:
            Validated dictionary matching response_schema
        
        Raises:
            ValueError: If response cannot be validated after retries
        """
        logger.info(f"[LLM] Calling {self.model}")
        
        # TODO: Implement actual API call (OpenAI, Anthropic, etc.)
        # For now, this is a stub that validates the schema
        
        # This will be filled in by the implementer based on their LLM choice
        # Pattern: 1) Call API with JSON schema 2) Parse response 3) Validate
        raise NotImplementedError(
            "call_json() requires API implementation. "
            "Set llm_provider in settings and implement the call."
        )
    
    def validate_json_response(self, response_text: str, schema: BaseModel) -> Dict[str, Any]:
        """
        Parse and validate LLM response against schema.
        
        Args:
            response_text: Raw API response
            schema: Pydantic model for validation
        
        Returns:
            Validated dict
        
        Raises:
            ValueError: If response is invalid
        """
        try:
            # Try to extract JSON from response (LLM sometimes wraps it in markdown)
            json_text = response_text.strip()
            if json_text.startswith("```"):
                json_text = json_text.split("```")[1].strip()
                if json_text.startswith("json"):
                    json_text = json_text[4:].strip()
            
            data = json.loads(json_text)
            # Validate against schema
            validated = schema(**data)
            return validated.dict()
        except Exception as e:
            logger.error(f"[LLM] Failed to validate response: {e}")
            raise ValueError(f"Invalid LLM response: {e}")


# Singleton instance (lazy-loaded)
_llm_client: Optional[LLMClient] = None


def get_llm_client() -> LLMClient:
    """Get or create LLM client from settings."""
    global _llm_client
    if _llm_client is None:
        from app.core.settings import settings
        if not settings.llm_api_key:
            raise ValueError(
                "LLM API key not configured. "
                "Set LLM_API_KEY in .env or environment."
            )
        _llm_client = LLMClient(
            api_key=settings.llm_api_key,
            model=settings.llm_model,
        )
    return _llm_client
```
